# 36장 디스트럭처링 할당
디스트럭처링 할당(구조 분해 할당)은 구조화된 배열,객체를 비구조화 하여 변수에 할당하는것을 말한다
## 36.1 배열 디스트릭처링 할당
```javascript
var arr = [1, 2, 3];

var [one, two, three] = arr; 

console.log(one);   // 1
console.log(two);   // 2
console.log(three); // 3
// 우변의 배열 요소를 좌변의 변수 이름에 순서대로 할당한다 (인덱스 기준)

const [a,b] = {} // TypeError
// 우변에 이터러블을 할당하지 않으면 에러가 발생

const [c,d] = [1]
console.log(c,d) // 1 undefined
// 변수의 개수와 이터러블의 요소 개수가 일치할 필요는 없다

const [e,f=10,g=3]  = [1,2]
console.log(e,f,g) // 1 2 3
// 기본값을 설정할 수 있다 (할당 값이 우선)

const [x,...y] = [1,2,3,4]
console.log(x,y) // 1 [2,3,4]
// Rest 요소를 사용할 수 있다 
```
## 36.2 객체 디스트럭처링 할당

```javascript
const user = {firstName:"Ungmo",lastName:"lee"}
const {lastName,firstName} = user
console.log(fistName,lastName) // Ungmo lee
// 할당문의 우변은 객체여야하고 할당 기준은 프로퍼티 키다(순서가 의미 없고 선언된 변수 이름과 프로퍼티 키만 일치하면 할당됨)

const {lastName,firstName} // SyntaxError
// 우변에 객체 또는 객체로 평가 될수 있는 표현식이 아니면 에러 발생

const {lastName,firstName} = user
const {lastName:ln,firstName:fn} = user
console.log(ln,fn) // Ungmo lee
// 축약 표현이 사용가능하다 


const { lastName, firstName = 'Ungmo' } = { lastName: "lee" };
console.log(firstName, lastName);  // Ungmo lee

const { lastName: ln, firstName: fn = 'Ungmo' } = { lastName: 'lee' };
console.log(fn, ln);  // Ungmo lee
// 변수에 기본 값 설정 가능

// 객체 디스트럭처링 할당은 객체에서 프로퍼티 키로 필요한 프로퍼티 값만 추출하여 변수에 할당하ㅗ 싶을 때 유용함

// 중첩 객체 디스트럭처링
const user = {
  name: 'Lee',
  address: {
    zipCode: '03068',
    city: 'Seoul'
  }
};

const { address: { city } } = user;
console.log(city); // Seoul
// 중첩 객체의 프로퍼티를 꺼낼 때는 프로퍼티 키를 중간 경로로 사용하고
// 실제 값만 변수에 바인딩된다. address 자체는 변수로 선언되지 않는다.

const {x,...rest} = {x:1,y:2,z:3}
console.log(x,rest)
// Rest 프로퍼티 사용할 수 있다
```
